NumPy
Numerical Python - Foundation of Scientific Computing
NumPy provides fast N-dimensional arrays and math operations. It is 50-100x faster than Python lists because operations run in C. Every scientific Python library (Pandas, TensorFlow, OpenCV) is built on NumPy.
pip install numpy
ArraysMathLinear AlgebraStatisticsBroadcasting
Used by:NASAOpenAIPandasTensorFlowscikit-learn

🧠 Core Concepts

ndarray
N-dimensional array. All elements same type. Stored in contiguous memory for speed.
dtype
Data type: int32, float64, bool. Determines memory and operations.
shape
Dimensions: (5,)=1D, (3,4)=2D matrix, (2,3,4)=3D tensor.
Broadcasting
arr+5 adds 5 to EVERY element. No loop needed.
Vectorization
Apply operations to entire array at once. 50x faster than loops.
Boolean Indexing
arr[arr>5] returns only elements >5. Powerful filtering.
reshape()
Change array shape: (12,) -> (3,4) without copying data.
np.linalg
Linear algebra: dot product, inverse, determinant, eigenvalues.

from flask import Flask, request, jsonify

app = Flask(__name__)
todos = []
next_id = 1

@app.route('/api/todos', methods=['GET'])
def get_todos():
    return jsonify({'todos': todos, 'count': len(todos)})

@app.route('/api/todos', methods=['POST'])
def add_todo():
    global next_id
    data = request.json
    todo = {'id': next_id, 'task': data['task'], 'done': False}
    todos.append(todo)
    next_id += 1
    return jsonify(todo), 201

@app.route('/api/todos/', methods=['PUT'])
def update_todo(todo_id):
    todo = next((t for t in todos if t['id']==todo_id), None)
    if not todo:
        return jsonify({'error': 'Not found'}), 404
    todo['done'] = request.json.get('done', todo['done'])
    return jsonify(todo)

@app.route('/api/todos/', methods=['DELETE'])
def delete_todo(todo_id):
    global todos
    todos = [t for t in todos if t['id'] != todo_id]
    return jsonify({'message': 'Deleted'})

if __name__ == '__main__':
    app.run(debug=True)
Project 2: Simple Calculator API
REST API for math operations.
python
from flask import Flask, request, jsonify

app = Flask(__name__)

@app.route('/api/calc', methods=['POST'])
def calculate():
    data = request.json
    a = data.get('a', 0)
    b = data.get('b', 0)
    op = data.get('op', '+')
    
    ops = {
        '+': lambda x,y: x+y,
        '-': lambda x,y: x-y,
        '*': lambda x,y: x*y,
        '/': lambda x,y: x/y if y!=0 else None
    }
    
    if op not in ops:
        return jsonify({'error': 'Invalid operator'}), 400
    
    result = ops[op](a, b)
    if result is None:
        return jsonify({'error': 'Division by zero'}), 400
    
    return jsonify({'a': a, 'b': b, 'op': op, 'result': result})

if __name__ == '__main__':
    app.run(debug=True)

from django.db import models

class Post(models.Model):
    title = models.CharField(max_length=200)
    content = models.TextField()
    author = models.CharField(max_length=100)
    created_at = models.DateTimeField(auto_now_add=True)
    published = models.BooleanField(default=False)
    
    def __str__(self):
        return self.title
    
    class Meta:
        ordering = ['-created_at']

class Comment(models.Model):
    post = models.ForeignKey(Post, on_delete=models.CASCADE, related_name='comments')
    text = models.TextField()
    author = models.CharField(max_length=100)
    created_at = models.DateTimeField(auto_now_add=True)

# views.py
from django.shortcuts import render, get_object_or_404
from django.http import JsonResponse
from .models import Post

def post_list(request):
    posts = Post.objects.filter(published=True).order_by('-created_at')
    return render(request, 'blog/post_list.html', {'posts': posts})

def post_detail(request, pk):
    post = get_object_or_404(Post, pk=pk)
    comments = post.comments.all()
    return render(request, 'blog/post_detail.html', {'post': post, 'comments': comments})

# API view
def api_posts(request):
    posts = Post.objects.filter(published=True).values('id','title','author','created_at')
    return JsonResponse({'posts': list(posts)})

Project: Face Detector
Detect faces in images using Haar Cascades.
python
import cv2
import numpy as np

# Create a simple test image with a face-like pattern
img = np.zeros((400, 400, 3), dtype=np.uint8)
img[:] = (30, 30, 50)  # dark background

# Draw a simple face
cv2.circle(img, (200, 180), 100, (200, 180, 150), -1)  # face
cv2.circle(img, (165, 155), 15, (50, 50, 50), -1)      # left eye
cv2.circle(img, (235, 155), 15, (50, 50, 50), -1)      # right eye
cv2.ellipse(img, (200, 210), (40, 20), 0, 0, 180, (150, 80, 80), -1)  # mouth

# Convert to grayscale
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# Load face detector
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')

# Detect faces
faces = face_cascade.detectMultiScale(gray, 1.1, 4)
print(f"Detected {len(faces)} face(s)")

# Draw rectangles
for (x, y, w, h) in faces:
    cv2.rectangle(img, (x,y), (x+w,y+h), (0,255,0), 2)
    cv2.putText(img, 'Face', (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0,255,0), 1)

cv2.imshow('Face Detection', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
